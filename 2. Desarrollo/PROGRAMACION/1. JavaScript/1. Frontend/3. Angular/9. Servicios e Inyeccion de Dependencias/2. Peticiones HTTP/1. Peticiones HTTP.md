## Pasos Previos:
- Antes de realizar una peticion HTTP deberemos configurar en nuestro `{ts icon} main.ts` el siguiente modulo.
- Si no lo importamos dara un Null Injection Error
	
	```ts title:app.config.ts
	import { ApplicationConfig, importProvidersFrom, provideZoneChangeDetection } from '@angular/core';
	import { provideRouter } from '@angular/router';
	import { routes } from './app.routes';
	
	//  Importar esto
	import { provideHttpClient, withFetch } from '@angular/common/http';
	
	
	export const appConfig: ApplicationConfig = {
	  providers: [
	    provideZoneChangeDetection({ eventCoalescing: true }), 
	    provideRouter(routes), 
	    provideClientHydration(withEventReplay()),
	    provideHttpClient(withFetch()), //  Solo es necesario importar estos elementos
	  ]
	};
	```



### Http Client:

- **Servicio de Peticiones:**
	```ts
	// api.service.ts
	import { Injectable } from '@angular/core';
	import { HttpClient, HttpHeaders } from '@angular/common/http';
	import { Observable } from 'rxjs';
	
	@Injectable({
	  providedIn: 'root'
	})
	export class ApiService {
	  private apiUrl = 'https://api.example.com/items'; // URL base de la API
	
	  constructor(private http: HttpClient) {}
	
	  // M茅todo GET: Obtener una lista de elementos
	  getItems(): Observable<any[]> {
	    return this.http.get<any[]>(this.apiUrl); // Realiza la petici贸n GET
	  }
	
	  // M茅todo POST: Crear un nuevo elemento
	  createItem(item: any): Observable<any> {
	    return this.http.post<any>(this.apiUrl, item); // Realiza la petici贸n POST
	  }
	
	  // M茅todo PATCH: Actualizar un elemento existente
	  updateItem(id: number, item: any): Observable<any> {
	    return this.http.patch<any>(`${this.apiUrl}/${id}`, item); // Realiza la petici贸n PATCH
	  }
	
	  // M茅todo DELETE: Eliminar un elemento
	  deleteItem(id: number): Observable<any> {
	    return this.http.delete<any>(`${this.apiUrl}/${id}`); // Realiza la petici贸n DELETE
	  }
	}
	```
