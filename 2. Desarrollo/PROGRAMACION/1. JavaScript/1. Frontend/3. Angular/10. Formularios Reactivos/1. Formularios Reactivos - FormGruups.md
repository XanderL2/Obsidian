## Formularios Reactivos:
- Para usarlos debemos poner en los imports `{ts icon} ReactiveFormsModule`.
- 

### Creacion de Formularios Reactivos:
- **HTML:**

	```ts
	import { compileNgModule } from '@angular/compiler';
	import { Component, OnInit } from '@angular/core';
	import { FormControl, FormGroup, ReactiveFormsModule } from '@angular/forms';
	
	@Component({
	  selector: 'app-task-form',
	  standalone: true,
	  imports: [ReactiveFormsModule], // ðŸ‘ˆ Debemos importar el componente
	  template: `
	
	  `,
	  styles: ``
	})
	export class TaskFormComponent implements OnInit {
	
	  taskForm!: FormGroup;
	
	  ngOnInit(): void {
	    this.taskForm = new FormGroup({
	      taskName: new FormControl(""), // ðŸ‘ˆ Valores por defecto
	      description: new FormControl("") // ðŸ‘ˆ Valores por defecto
	    });
	  }
	
	//	ðŸ‘‡ Funcio para manejar la subida del form
	  onSubmit(){
	    console.log(this.taskForm.value);
	  }  
	
	}
	```

- **HTML:**
	```html

	
    <div>
	Nombre de nuestro objeto ðŸ‘‡      
      <form [formGroup]="taskForm" (ngSubmit)="onSubmit()"> ðŸ‘ˆ Param manejar la subida
				
        <label for="taskName">Task Name:</label>
        <input id="taskName" formControlName="taskName"> ðŸ‘ˆ Vinculamos con lo esteblecido

        <label for="description">Description:</label>
        <input id="description" formControlName="description">

        <button type="submit">Add</button>

      </form>
    </div>
	```
